/* Include the standard input / output functions */
#include <stdio.h>

void doors (int n)
{
	/* Initialize the array of doors to 0 (closed) */
	char is_open[n] = {0};
	int pass, door;

	/* Process the doors */
	for (pass = 0; pass < n; ++pass)
		for (door = pass; door < n; door += pass+1)
			is_open[door] = !is_open[door];

	/* Print out the results */
	for (door = 0; door < n; ++door)
		printf("Door #%d is %s.\n", door+1, (is_open[door] ? "open" : "closed"));
}

int main()
{
	/* Call the doors function with n = 100 */
	doors(100);
	
	return 0;
}
