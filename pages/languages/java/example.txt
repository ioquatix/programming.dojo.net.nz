public class Doors {
	public static void main(String[] args) {
		boolean[] doors = new boolean[100];
		for(int pass = 0; pass < 100; pass++){
			for(int door = pass; door < 100; door += pass + 1){
				doors[door] = !doors[door];
			}
		}
		for(int i = 0; i < 100; i++){
		 	System.out.println("Door #" + (i + 1) + " is " + (doors[i] ? 
		 				"open." : "closed."));
		}
	}
}
