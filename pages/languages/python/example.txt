#!/usr/bin/env python

def doors (n):
	# Initialize an array
	doors = [False] * 100
	
	# Process the doors
	for i in range(100):
		for j in range(i, 100, i+1):
			doors[j] = not doors[j]
	
	# Print out the results
	for k, door in enumerate(doors):
		print "Door %d is" % (k+1), 'open.' if door else 'closed.'

# Call the function doors with n = 100
doors(100)
