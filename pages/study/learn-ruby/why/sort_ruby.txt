#	The run from the command line:
#		$ ruby sort_ruby.rb
#		
#	Outputs:
#		Name: Bob Age: 10
#		Name: Sally Age: 13
#		Name: John Age: 15

items = [
	{:name => "Bob", :age => 10},
	{:name => "John", :age => 15},
	{:name => "Sally", :age => 13}
]

sorted_by_age = items.sort_by{|item| item[:age]}

sorted_by_age.each do |item|
	puts "Name: #{item[:name]} Age: #{item[:age]}"
end
