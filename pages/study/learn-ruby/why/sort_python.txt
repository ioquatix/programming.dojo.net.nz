#	The run from the command line:
#		$ python sort_python.py
#		
#	Outputs:
#		Name: Bob Age: 10
#		Name: Sally Age: 13
#		Name: John Age: 15

items = [
	{'name': "Bob", 'age': 10},
	{'name': "John", 'age': 15},
	{'name': "Sally", 'age': 13}
]

sorted_by_age = sorted(items, lambda x, y: x['age'] - y['age'])

for p in sorted_by_age:
	print "Name: %s Age: %d" % (p['name'], p['age'])
